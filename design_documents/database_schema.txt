# Database Schema

## Users Table
- id: UUID (Primary Key)
- email: VARCHAR (unique)
- name: VARCHAR
- oauth_provider: VARCHAR
- oauth_id: VARCHAR (unique)
- created_at: TIMESTAMP
- updated_at: TIMESTAMP

## Households Table
- id: UUID (Primary Key)
- name: VARCHAR
- created_at: TIMESTAMP

## Household_Members Table
- user_id: UUID (Foreign Key -> Users.id)
- household_id: UUID (Foreign Key -> Households.id)
- role: VARCHAR
- joined_at: TIMESTAMP

## Chores Table
- id: UUID (Primary Key)
- household_id: UUID (Foreign Key -> Households.id)
- title: VARCHAR
- description: TEXT
- time_estimate: INTEGER
- frequency: VARCHAR
- assigned_to: UUID (Foreign Key -> Users.id, nullable)
- status: VARCHAR
- created_at: TIMESTAMP
- updated_at: TIMESTAMP
- due_date: DATE

## Notifications Table
- id: UUID (Primary Key)
- user_id: UUID (Foreign Key -> Users.id)
- chore_id: UUID (Foreign Key -> Chores.id, nullable)
- type: VARCHAR
- message: TEXT
- sent_at: TIMESTAMP

## UserPreferences Table
- id: UUID (Primary Key)
- user_id: UUID (Foreign Key -> Users.id, unique)
- notification_preferences: JSON
- chore_preferences: JSON
- theme: VARCHAR
- created_at: TIMESTAMP
- updated_at: TIMESTAMP

## Badges Table
- id: UUID (Primary Key)
- name: VARCHAR
- description: VARCHAR
- icon_url: VARCHAR

## UserBadges Table
- user_id: UUID (Foreign Key -> Users.id)
- badge_id: UUID (Foreign Key -> Badges.id)
- earned_at: TIMESTAMP

## CalendarIntegrations Table
- id: UUID (Primary Key)
- user_id: UUID (Foreign Key -> Users.id, unique)
- provider: VARCHAR
- access_token: VARCHAR
- refresh_token: VARCHAR
- expires_at: TIMESTAMP

## ChoreTemplates Table
- id: UUID (Primary Key)
- title: VARCHAR
- description: TEXT
- time_estimate: INTEGER
- frequency: VARCHAR
- created_at: TIMESTAMP
- updated_at: TIMESTAMP

## ChoreActivity Table
- id: UUID (Primary Key)
- chore_id: UUID (Foreign Key -> Chores.id)
- user_id: UUID (Foreign Key -> Users.id)
- action: VARCHAR
- timestamp: TIMESTAMP
